Original Python code used for this console demo 
tasks = []

def show_menu():
    print("1. Add Task")
    print("2. View Tasks")
    print("3. Remove Task")
    print("4. Exit")

def add_task():
    task = input("Enter new task: ").strip()
    if task:
        tasks.append(task)
        print(f"Task added: {task}")
    else:
        print("No task entered.")

def view_tasks():
    if not tasks:
        print("No tasks yet.")
    else:
        for i, t in enumerate(tasks, start=1):
            print(f"{i}. {t}")

def remove_task():
    if not tasks:
        print("No tasks to remove.")
        return
    try:
        num = int(input("Enter task number to remove: "))
        idx = num - 1
        if 0 <= idx < len(tasks):
            removed = tasks.pop(idx)
            print(f"Removed: {removed}")
        else:
            print("Invalid task number.")
    except ValueError:
        print("Please enter a valid integer.")

def main():
    print('Python 3.11.5 (default, Sep 2025)')
    print('Type "help", "credits", or "license" for more information.')
    print(">>> Welcome to the To Do List Console")
    while True:
        show_menu()
        choice = input(">>> ").strip().lower()
        if choice in ("help",):
            print("Help: Enter 1 to add, 2 to view, 3 to remove, 4 to exit.")
        elif choice in ("license",):
            print("License: MIT License. Use freely with attribution.")
        elif choice in ("credits",):
            print("Credits: Built by Mohammed as a guided project.")
        elif choice in ("version",):
            print("Python 3.11.5")
        elif choice in ("clear",):
            print("\n" * 50)
        elif choice in ("exit", "quit", "4"):
            print("Exiting...")
            break
        elif choice in ("1",):
            add_task()
        elif choice in ("2",):
            view_tasks()
        elif choice in ("3",):
            remove_task()
        else:
            print("Invalid choice. Enter 1, 2, 3, or 4.")

if __name__ == "__main__":
    main()
